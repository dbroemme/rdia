def render(override_size = nil)
    size_to_draw = @data_point_size
    if override_size 
        size_to_draw = override_size 
    end
    half_size = size_to_draw / 2
    Gosu::draw_rect(@x - half_size, @y - half_size,
                    size_to_draw, size_to_draw,
                    graphics_color, z_order) 
end
